- Create a new project called visits with a package.json ( with express * & redis 2.8.0 as dependencies) and with an index.js
- For the index.js
    const express = erequire('express');
    const redis = require('redis');

    const app = express();
    const client = redis.createClient();

    client.set('visits', 0);

    app.get('/', (req, res) => {
        client.get('visits, (errpr, visits) => {
            res.send('number of visits is ' + visits );
            client.set('visits', parseInt(visits) + 1)
        })
    })

    app.listen('8081, () => {
        console.log('Listening on port 8081')
    })
